54  inline void getAllEntryNames(std::vector<std::string>& names) const
55  {
56  names.clear();
57  for (auto& it : entries_)
58  {
59  names.push_back(it.first);
60  }
61  }
62  inline size_t getNumberOfEntries() const { return entries_.size(); }
63  inline bool getAllowedCollision(const std::string& name1, const std::string& name2) const
64  {
65  auto it = entries_.find(name1);
66  if (it == entries_.end()) return true;
67  return it->second.find(name2) == it->second.end();
68  }

70 private:
71  std::unordered_map<std::string, std::unordered_set<std::string>> entries_;
72 };
