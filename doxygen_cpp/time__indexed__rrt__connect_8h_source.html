230  double forwardTimeDistance(const Motion *a, const Motion *b) const
231  {
232  static const Eigen::VectorXd max_vel = si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->prob_->vel_limits;
233 
234  const OMPLTimeIndexedRNStateSpace::StateType *sa = static_cast<const OMPLTimeIndexedRNStateSpace::StateType *>(a->state);
235  const OMPLTimeIndexedRNStateSpace::StateType *sb = static_cast<const OMPLTimeIndexedRNStateSpace::StateType *>(b->state);
236 
237  double ta, tb;
238  Eigen::VectorXd qa, qb;
239  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->OMPLToExoticaState(a->state, qa, ta);
240  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->OMPLToExoticaState(b->state, qb, tb);
241 
242  if (tb < ta) return 1e10;
243  Eigen::VectorXd diff = (qb - qa).cwiseAbs();
244  double min_dt = (diff.array() / max_vel.array()).maxCoeff();
245  if (fabs(tb - ta) < min_dt) return 1e10;
246  return si_->distance(a->state, b->state);
247  }

268  bool correctTime(const Motion *a, Motion *b, bool reverse, bool &changed) const
269  {
270  Eigen::VectorXd max_vel = si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->prob_->vel_limits;
271  double ta, tb;
272  Eigen::VectorXd qa, qb;
273  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->OMPLToExoticaState(a->state, qa, ta);
274  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->OMPLToExoticaState(b->state, qb, tb);
275  Eigen::VectorXd diff = (qb - qa).cwiseAbs();
276  double min_dt = (diff.array() / max_vel.array()).maxCoeff();
277  if (fabs(tb - ta) < min_dt)
278  {
279  if (reverse_check_) return false;
280  tb = ta + (reverse ? -min_dt : min_dt);
281  changed = true;
282  }
283  else
284  changed = false;
285  if (tb < si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->prob_->t_start || tb > si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->prob_->GetGoalTime()) return false;
286  si_->getStateSpace()->as<OMPLTimeIndexedRNStateSpace>()->ExoticaToOMPLState(qb, tb, b->state);
287  return true;
288  }
