142  auto wsfront = std::find_if_not(s.begin(), s.end(), [](int c) { return std::isspace(c); });
143  return std::string(wsfront, std::find_if_not(s.rbegin(), std::string::const_reverse_iterator(wsfront), [](int c) { return std::isspace(c); }).base());

164 template <typename T, const int S> // Eigen::Vector<S><T>
165 inline Eigen::Matrix<T, S, 1> ParseVector(const std::string value)
166 {
167  Eigen::Matrix<T, S, 1> ret;
168  std::string temp_entry;
169  int i = 0;
170 
171  std::istringstream text_parser(value);
172 
173  while (text_parser >> temp_entry)
174  {
175  ret.conservativeResize(++i);
176  try
177  {
178  ret[i - 1] = ToNumber<T>(temp_entry);
179  }
180  catch (std::invalid_argument)
181  {
182  ret[i - 1] = std::numeric_limits<T>::quiet_NaN();
183  }
184  }
185  if (i == 0) ThrowPretty("Empty vector!");
186  if (S != Eigen::Dynamic && S != i)
187  {
188  ThrowPretty("Wrong vector size! Requested: " + std::to_string(S) + ", Provided: " + std::to_string(i));
189  }
190  return ret;
191 }
