75  void add(const Eigen::Ref<const Eigen::VectorXd>& x)
76  {
77  if (W == 0.)
78  {
79  W = 1.;
80  T = x;
81  S.setZero();
82  return;
83  }
84  W += 1.;
85  T += x;
86  double f = 1. / W / (W - 1.);
87  dX = W * x - T;
88  for (int r = 0; r < D; ++r)
89  {
90  for (int c = 0; c < D; ++c)
91  {
92  S(r, c) += f * dX(r) * dX(c);
93  }
94  }
95  }

102  void add(double& W_, const Eigen::Ref<const Eigen::VectorXd>& T_,
103  const Eigen::Ref<const Eigen::VectorXd>& S_)
104  {
105  if (W == 0.)
106  {
107  W = W_;
108  T = T_;
109  S = S_;
110  return;
111  }
112  dX = T_ / W_ - T / W;
113 
114  double f = W * W_ / (W + W_);
115  for (int r = 0; r < D; ++r)
116  {
117  for (int c = 0; c < D; ++c)
118  {
119  S(r, c) += S_(r, c) + f * dX(r) * dX(c);
120  }
121  }
122  T += T_;
123  W += W_;
124  }

126  inline void addw(double w, const Eigen::Ref<const Eigen::VectorXd>& x)
127  {
128  if (W == 0.)
129  {
130  W = w;
131  T = w * x;
132  S.setZero();
133  return;
134  }
135 
136  dX = x - T / W;
137 
138  double f = W * w / (W + w);
139  for (int r = 0; r < D; ++r)
140  {
141  for (int c = 0; c < D; ++c)
142  {
143  S(r, c) += f * dX(r) * dX(c);
144  }
145  }
146 
147  T += w * x;
148  W += w;
149  }
